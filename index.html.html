  <script>

    let points = 50000;

    let role = '';

    let name = '';

    function startGame() {

      name = document.getElementById('name').value || "Khoa";

      role = document.getElementById('roleSelect').value;

     

      document.getElementById('login').classList.add('hidden');

      document.getElementById('main').classList.remove('hidden');

     

      document.getElementById('header').innerHTML = `

        <p>Chào <span class="text-green-400">\( {name}</span> ( \){role})</p>

        <p class="text-3xl font-bold text-yellow-400">${points.toLocaleString('vi-VN')} Điểm</p>

      `;

      loadContent();

    }

    function loadContent() {

      let html = `

        <h2 class="text-2xl font-bold mb-6">Bóng Đá Ảo</h2>

        <div class="space-y-6">

          <div class="bg-zinc-900 p-6 rounded-3xl">

            <p class="text-yellow-400">Premier League Ảo</p>

            <h3>Man City vs Liverpool</h3>

            <div class="grid grid-cols-3 gap-3 mt-4">

              <button onclick="bet(1.85, 'Man City')" class="bg-green-600 py-6 rounded-2xl">1.85</button>

              <button onclick="bet(3.6, 'Hòa')" class="bg-yellow-600 py-6 rounded-2xl">3.60</button>

              <button onclick="bet(4.1, 'Liverpool')" class="bg-red-600 py-6 rounded-2xl">4.10</button>

            </div>

          </div>

          <!-- Thêm nhiều trận khác tương tự nếu cần -->

        </div>

        <div class="grid grid-cols-2 gap-4 mt-10">

          <button onclick="casino()" class="bg-purple-600 py-12 rounded-3xl text-xl">🎰 Casino</button>

          <button onclick="fish()" class="bg-teal-600 py-12 rounded-3xl text-xl">🐟 Bắn Cá</button>

        </div>

        ${role === 'admin' ? '<button onclick="adminPanel()" class="w-full mt-8 bg-violet-600 py-6 rounded-3xl font-bold">👑 Admin Panel</button>' : ''}

        ${role === 'agent' ? '<button onclick="agentPanel()" class="w-full mt-8 bg-orange-600 py-6 rounded-3xl font-bold">👥 Đại Lý Panel</button>' : ''}

      `;

      document.getElementById('content').innerHTML = html;

    }

    function bet(odds, team) {

      const amount = parseInt(prompt(`Cược ${team} (tỉ lệ ${odds})\nNhập số điểm:`, "1000"));

      if (amount && amount <= points) {

        points -= amount;

        const win = Math.round(amount * odds);

        alert(`✅ Cược ${team} thành công!\nNếu thắng nhận ${win} điểm!`);

        document.getElementById('header').querySelector('p:last-child').textContent = points.toLocaleString('vi-VN') + " Điểm";

      }

    }

    function casino() { alert("🎰 Casino đang phát triển..."); }

    function fish() { alert("🐟 Bắn Cá đang phát triển..."); }

    function adminPanel() { alert("👑 Admin Panel - Khoa quản lý toàn bộ"); }

    function agentPanel() { alert("👥 Đại Lý - Quản lý thành viên"); }

    function logout() { location.reload(); }

  </script>
